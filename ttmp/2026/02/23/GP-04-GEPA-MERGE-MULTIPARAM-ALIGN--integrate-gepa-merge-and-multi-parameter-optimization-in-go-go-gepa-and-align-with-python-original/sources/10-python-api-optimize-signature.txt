    40	
    41	def optimize(
    42	    seed_candidate: dict[str, str],
    43	    trainset: list[DataInst] | DataLoader[DataId, DataInst],
    44	    valset: list[DataInst] | DataLoader[DataId, DataInst] | None = None,
    45	    adapter: GEPAAdapter[DataInst, Trajectory, RolloutOutput] | None = None,
    46	    task_lm: str | ChatCompletionCallable | None = None,
    47	    evaluator: Evaluator | None = None,
    48	    # Reflection-based configuration
    49	    reflection_lm: LanguageModel | str | None = None,
    50	    candidate_selection_strategy: CandidateSelector | Literal["pareto", "current_best", "epsilon_greedy"] = "pareto",
    51	    frontier_type: FrontierType = "instance",
    52	    skip_perfect_score: bool = True,
    53	    batch_sampler: BatchSampler | Literal["epoch_shuffled"] = "epoch_shuffled",
    54	    reflection_minibatch_size: int | None = None,
    55	    perfect_score: float = 1.0,
    56	    reflection_prompt_template: str | dict[str, str] | None = None,
    57	    custom_candidate_proposer: ProposalFn | None = None,
    58	    # Component selection configuration
    59	    module_selector: ReflectionComponentSelector | str = "round_robin",
    60	    # Merge-based configuration
    61	    use_merge: bool = False,
    62	    max_merge_invocations: int = 5,
    63	    merge_val_overlap_floor: int = 5,
    64	    # Budget and Stop Condition
    65	    max_metric_calls: int | None = None,
    66	    stop_callbacks: StopperProtocol | Sequence[StopperProtocol] | None = None,
    67	    # Logging and Callbacks
    68	    logger: LoggerProtocol | None = None,
    69	    run_dir: str | None = None,
    70	    callbacks: "list[GEPACallback] | None" = None,
    71	    use_wandb: bool = False,
    72	    wandb_api_key: str | None = None,
    73	    wandb_init_kwargs: dict[str, Any] | None = None,
    74	    use_mlflow: bool = False,
    75	    mlflow_tracking_uri: str | None = None,
    76	    mlflow_experiment_name: str | None = None,
    77	    track_best_outputs: bool = False,
    78	    display_progress_bar: bool = False,
    79	    use_cloudpickle: bool = False,
    80	    # Evaluation caching
    81	    cache_evaluation: bool = False,
    82	    # Reproducibility
    83	    seed: int = 0,
    84	    raise_on_exception: bool = True,
    85	    val_evaluation_policy: EvaluationPolicy[DataId, DataInst] | Literal["full_eval"] | None = None,
    86	) -> GEPAResult[RolloutOutput, DataId]:
    87	    """
    88	    GEPA is an evolutionary optimizer that evolves (multiple) text components of a complex system to optimize them towards a given metric.
    89	    GEPA can also leverage rich textual feedback obtained from the system's execution environment, evaluation,
    90	    and the system's own execution traces to iteratively improve the system's performance.
    91	
    92	    Concepts:
    93	    - System: A harness that uses text components to perform a task. Each text component of the system to be optimized is a named component of the system.
    94	    - Candidate: A mapping from component names to component text. A concrete instantiation of the system is realized by setting the text of each system component
    95	      to the text provided by the candidate mapping.
    96	    - `DataInst`: An (uninterpreted) data type over which the system operates.
    97	    - `RolloutOutput`: The output of the system on a `DataInst`.
    98	
    99	    Each execution of the system produces a `RolloutOutput`, which can be evaluated to produce a score. The execution of the system also produces a trajectory,
   100	    which consists of the operations performed by different components of the system, including the text of the components that were executed.
   101	
   102	    GEPA can be applied to optimize any system that uses text components (e.g., prompts in a AI system, code snippets/code files/functions/classes in a codebase, etc.).
   103	    In order for GEPA to plug into your system's environment, GEPA requires an adapter, `GEPAAdapter` to be implemented. The adapter is responsible for:
   104	    1. Evaluating a proposed candidate on a batch of inputs.
   105	       - The adapter receives a candidate proposed by GEPA, along with a batch of inputs selected from the training/validation set.
   106	       - The adapter instantiates the system with the texts proposed in the candidate.
   107	       - The adapter then evaluates the candidate on the batch of inputs, and returns the scores.
   108	       - The adapter should also capture relevant information from the execution of the candidate, like system and evaluation traces.
   109	    2. Identifying textual information relevant to a component of the candidate
   110	       - Given the trajectories captured during the execution of the candidate, GEPA selects a component of the candidate to update.
   111	       - The adapter receives the candidate, the batch of inputs, and the trajectories captured during the execution of the candidate.
   112	       - The adapter is responsible for identifying the textual information relevant to the component to update.
   113	       - This information is used by GEPA to reflect on the performnace of the component, and propose new component texts.
   114	
   115	    At each iteration, GEPA proposes a new candidate using one of the following strategies:
   116	    1. Reflective mutation: GEPA proposes a new candidate by mutating the current candidate, leveraging rich textual feedback.
   117	    2. Merge: GEPA proposes a new candidate by merging 2 candidates that are on the Pareto frontier.
   118	
   119	    GEPA also tracks the Pareto frontier of performance achieved by different candidates on the validation set. This way, it can leverage candidates that
   120	    work well on a subset of inputs to improve the system's performance on the entire validation set, by evolving from the Pareto frontier.
   121	
   122	    Parameters:
   123	    - seed_candidate: The initial candidate to start with.
   124	    - trainset: Training data supplied as an in-memory sequence or a `DataLoader` yielding batches for reflective updates.
   125	    - valset: Validation data source (sequence or `DataLoader`) used for tracking Pareto scores. If not provided, GEPA reuses the trainset.
   126	    - adapter: A `GEPAAdapter` instance that implements the adapter interface. This allows GEPA to plug into your system's environment. If not provided, GEPA will use a default adapter: `gepa.adapters.default_adapter.default_adapter.DefaultAdapter`, with model defined by `task_lm`.
   127	    - task_lm: Optional. The model to use for the task. This is only used if `adapter` is not provided, and is used to initialize the default adapter.
   128	    - evaluator: Optional. A custom evaluator to use for evaluating the candidate program. If not provided, GEPA will use the default evaluator: `gepa.adapters.default_adapter.default_adapter.ContainsAnswerEvaluator`. Only used if `adapter` is not provided.
   129	
   130	    # Reflection-based configuration
   131	    - reflection_lm: A `LanguageModel` instance that is used to reflect on the performance of the candidate program.
   132	    - candidate_selection_strategy: The strategy to use for selecting the candidate to update. Supported strategies: 'pareto', 'current_best', 'epsilon_greedy'. Defaults to 'pareto'.
   133	    - frontier_type: Strategy for tracking Pareto frontiers. 'instance' tracks per validation example, 'objective' tracks per objective metric, 'hybrid' combines both, 'cartesian' tracks per (example, objective) pair. Defaults to 'instance'.
   134	    - skip_perfect_score: Whether to skip updating the candidate if it achieves a perfect score on the minibatch.
   135	    - batch_sampler: Strategy for selecting training examples. Can be a [BatchSampler](src/gepa/strategies/batch_sampler.py) instance or a string for a predefined strategy from ['epoch_shuffled']. Defaults to 'epoch_shuffled', which creates an [EpochShuffledBatchSampler](src/gepa/strategies/batch_sampler.py).
   136	    - reflection_minibatch_size: The number of examples to use for reflection in each proposal step. Defaults to 3. Only valid when batch_sampler='epoch_shuffled' (default), and is ignored otherwise.
   137	    - perfect_score: The perfect score to achieve.
   138	    - reflection_prompt_template: The prompt template to use for reflection. Can be either a string (applied to all components) or a dict mapping component names to their specific templates. If not provided, GEPA will use the default prompt template (see [InstructionProposalSignature](src/gepa/strategies/instruction_proposal.py)). Each prompt template must contain the following placeholders, which will be replaced with actual values: `<curr_param>` (will be replaced by the instructions/component to evolve) and `<side_info>` (replaced with the inputs, outputs, and feedback generated with current instruction). When using a dict, components without a specified template will use the default template. This will be ignored if the adapter provides its own `propose_new_texts` method.
   139	    - custom_candidate_proposer: Optional custom function for proposing new candidates. If provided, this will be used instead of the default LLM-based reflection approach. Cannot be used if adapter provides `propose_new_texts`. Signature: `(candidate, reflective_dataset, components_to_update) -> dict[str, str]`.
   140	
   141	    # Component selection configuration
   142	    - module_selector: Component selection strategy. Can be a ReflectionComponentSelector instance or a string ('round_robin', 'all'). Defaults to 'round_robin'. The 'round_robin' strategy cycles through components in order. The 'all' strategy selects all components for modification in every GEPA iteration.
   143	
   144	    # Merge-based configuration
   145	    - use_merge: Whether to use the merge strategy.
   146	    - max_merge_invocations: The maximum number of merge invocations to perform.
   147	    - merge_val_overlap_floor: Minimum number of shared validation ids required between parents before attempting a merge subsample. Only relevant when using `val_evaluation_policy` other than `full_eval`.
   148	
   149	    # Budget and Stop Condition
   150	    - max_metric_calls: Optional maximum number of metric calls to perform. If not provided, stop_callbacks must be provided.
   151	    - stop_callbacks: Optional stopper(s) that return True when optimization should stop. Can be a single StopperProtocol or a list or tuple of StopperProtocol instances. Examples: FileStopper, TimeoutStopCondition, SignalStopper, NoImprovementStopper, or custom stopping logic. If not provided, max_metric_calls must be provided.
   152	
   153	    # Logging and Callbacks
   154	    - logger: A `LoggerProtocol` instance that is used to log the progress of the optimization.
   155	    - callbacks: Optional list of callback objects for observing optimization progress. Callbacks receive events like on_optimization_start, on_iteration_start, on_candidate_accepted, etc. See `gepa.core.callbacks.GEPACallback` for the full protocol.
   156	    - run_dir: The directory to save the results to. Optimization state and results will be saved to this directory. If the directory already exists, GEPA will read the state from this directory and resume the optimization from the last saved state. If provided, a FileStopper is automatically created which checks for the presence of "gepa.stop" in this directory, allowing graceful stopping of the optimization process upon its presence.
   157	    - use_wandb: Whether to use Weights and Biases to log the progress of the optimization.
   158	    - wandb_api_key: The API key to use for Weights and Biases.
   159	    - wandb_init_kwargs: Additional keyword arguments to pass to the Weights and Biases initialization.
   160	    - use_mlflow: Whether to use MLflow to log the progress of the optimization.
   161	      Both wandb and mlflow can be used simultaneously if desired.
   162	    - mlflow_tracking_uri: The tracking URI to use for MLflow.
   163	    - mlflow_experiment_name: The experiment name to use for MLflow.
   164	    - track_best_outputs: Whether to track the best outputs on the validation set. If True, GEPAResult will contain the best outputs obtained for each task in the validation set.
   165	    - display_progress_bar: Show a tqdm progress bar over metric calls when enabled.
   166	    - use_cloudpickle: Use cloudpickle instead of pickle. This can be helpful when the serialized state contains dynamically generated DSPy signatures.
   167	
   168	    # Evaluation caching
   169	    - cache_evaluation: Whether to cache the (score, output, objective_scores) of (candidate, example) pairs. If True and a cache entry exists, GEPA will skip the fitness evaluation and use the cached results. This helps avoid redundant evaluations and saves metric calls. Defaults to False.
   170	
   171	    # Reproducibility
   172	    - seed: The seed to use for the random number generator.
   173	    - val_evaluation_policy: Strategy controlling which validation ids to score each iteration and which candidate is currently best. Supported strings: "full_eval" (evaluate every id each time) Passing None defaults to "full_eval".
   174	    - raise_on_exception: Whether to propagate proposer/evaluator exceptions instead of stopping gracefully.
   175	    """
   176	    # Validate seed_candidate is not None or empty
   177	    if seed_candidate is None or not seed_candidate:
   178	        raise ValueError("seed_candidate must contain at least one component text.")
   179	
   180	    active_adapter: GEPAAdapter[DataInst, Trajectory, RolloutOutput] | None = None
   181	    if adapter is None:
   182	        assert task_lm is not None, (
   183	            "Since no adapter is provided, GEPA requires a task LM to be provided. Please set the `task_lm` parameter."
   184	        )
   185	        active_adapter = cast(
   186	            GEPAAdapter[DataInst, Trajectory, RolloutOutput], DefaultAdapter(model=task_lm, evaluator=evaluator)
   187	        )
   188	    else:
   189	        assert task_lm is None, (
   190	            "Since an adapter is provided, GEPA does not require a task LM to be provided. Please set the `task_lm` parameter to None."
   191	        )
   192	        assert evaluator is None, (
   193	            "Since an adapter is provided, GEPA does not require an evaluator to be provided. Please set the `evaluator` parameter to None."
   194	        )
   195	        active_adapter = adapter
   196	
   197	    # Normalize datasets to DataLoader instances
   198	    train_loader = ensure_loader(trainset)
   199	    val_loader = ensure_loader(valset) if valset is not None else train_loader
   200	
   201	    # Comprehensive stop_callback logic
   202	    # Convert stop_callbacks to a list if it's not already
   203	    stop_callbacks_list: list[StopperProtocol] = []
   204	    if stop_callbacks is not None:
   205	        if isinstance(stop_callbacks, Sequence):
   206	            stop_callbacks_list.extend(stop_callbacks)
   207	        else:
   208	            stop_callbacks_list.append(stop_callbacks)
   209	
   210	    # Add file stopper if run_dir is provided
   211	    if run_dir is not None:
   212	        stop_file_path = os.path.join(run_dir, "gepa.stop")
   213	        file_stopper = FileStopper(stop_file_path)
   214	        stop_callbacks_list.append(file_stopper)
   215	
   216	    # Add max_metric_calls stopper if provided
   217	    if max_metric_calls is not None:
   218	        from gepa.utils import MaxMetricCallsStopper
   219	
   220	        max_calls_stopper = MaxMetricCallsStopper(max_metric_calls)
