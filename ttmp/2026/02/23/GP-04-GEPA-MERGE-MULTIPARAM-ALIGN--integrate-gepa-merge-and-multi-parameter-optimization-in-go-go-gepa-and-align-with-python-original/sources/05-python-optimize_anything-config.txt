   841	def make_litellm_lm(model_name: str) -> LanguageModel:
   842	    """Convert a LiteLLM model name string to a :class:`LanguageModel` callable.
   843	
   844	    The returned callable conforms to the ``LanguageModel`` protocol and
   845	    accepts a plain ``str`` prompt, a ``list[dict]`` chat-messages list, or
   846	    a multimodal messages list (with content arrays containing images).
   847	    """
   848	    import litellm
   849	
   850	    def _lm(prompt: str | list[dict[str, Any]]) -> str:
   851	        if isinstance(prompt, str):
   852	            messages: list[dict[str, Any]] = [{"role": "user", "content": prompt}]
   853	        else:
   854	            messages = prompt
   855	        completion = litellm.completion(model=model_name, messages=messages)
   856	        return completion.choices[0].message.content  # type: ignore[union-attr]
   857	
   858	    return _lm

   861	class EvaluatorWrapper:
   862	    """Internal wrapper that adapts a user's evaluator to GEPA's internal interface.
   863	
   864	    Handles: single-instance mode (omits ``example``), str-candidate unwrapping,
   865	    kwarg filtering (incl. ``opt_state`` injection), ``oa.log()`` capture,
   866	    optional stdout/stderr capture, and normalizing the return value to
   867	    ``(score, output, side_info)`` regardless of what the user returns.
   868	    """
   869	
   870	    def __init__(
   871	        self,
   872	        evaluator_fn: Callable[..., Any],
   873	        single_instance_mode: bool,
   874	        capture_stdio: bool = False,
   875	        str_candidate_mode: bool = False,
   876	        raise_on_exception: bool = True,
   877	    ) -> None:
   878	        # Inspect the evaluator's signature once to determine which kwargs it accepts.
   879	        sig = inspect.signature(evaluator_fn)
   880	        has_var_keyword = any(p.kind == inspect.Parameter.VAR_KEYWORD for p in sig.parameters.values())
   881	        if has_var_keyword:
   882	            accepted_params = None  # accept all
   883	        else:
   884	            accepted_params = set(sig.parameters.keys())
   885	
   886	        def _filter_kwargs(kwargs: dict) -> dict:
   887	            if accepted_params is None:
   888	                return kwargs
   889	            return {k: v for k, v in kwargs.items() if k in accepted_params}
   890	
   891	        def wrapped_evaluator(
   892	            candidate: Candidate, example: object | None = None, **kwargs: Any
   893	        ) -> tuple[float, Any, SideInfo]:
   894	            # Create a fresh, shared log context for this evaluator call.
   895	            # The same LogContext is accessible from child threads via
   896	            # oa.get_log_context() / oa.set_log_context().
   897	            log_ctx = LogContext()
   898	            _set_log_context(log_ctx)
   899	
   900	            # Build full kwargs dict. In single-instance mode, don't forward
   901	            # example to the evaluator at all.
   902	            if single_instance_mode:
   903	                all_kwargs = kwargs
   904	            else:
   905	                all_kwargs = {"example": example, **kwargs}
   906	
   907	            filtered = _filter_kwargs(all_kwargs)
   908	
   909	            # Unwrap candidate for str_candidate_mode
   910	            eval_candidate: Candidate | str = candidate
   911	            if str_candidate_mode:
   912	                eval_candidate = candidate[_STR_CANDIDATE_KEY]
   913	
   914	            # Acquire per-thread stream capture from the shared manager per-call.
   915	            # This scopes the sys.stdout/stderr replacement to only the duration
   916	            # of evaluator execution, restoring the originals between calls.
   917	            # Both acquire/start_capture and the evaluator call are inside the
   918	            # same try/finally so that stream_manager.release() is always called
   919	            # even if start_capture() raises (e.g. assertion on double-capture).
   920	            stdout_capturer: ThreadLocalStreamCapture | None = None
   921	            stderr_capturer: ThreadLocalStreamCapture | None = None
   922	            try:
   923	                if capture_stdio:
   924	                    stdout_capturer, stderr_capturer = stream_manager.acquire()
   925	                    stdout_capturer.start_capture()
   926	                    stderr_capturer.start_capture()
   927	
   928	                result = evaluator_fn(eval_candidate, **filtered)
   929	            except Exception as e:
   930	                result = e  # Sentinel; handled below after cleanup
   931	            finally:
   932	                captured_stdout = stdout_capturer.stop_capture() if stdout_capturer else ""
   933	                captured_stderr = stderr_capturer.stop_capture() if stderr_capturer else ""
   934	                if capture_stdio and stdout_capturer is not None:
   935	                    stream_manager.release()
   936	                log_output = log_ctx.drain()
   937	                _set_log_context(None)
   938	
   939	            # If evaluator raised, preserve captured diagnostics
   940	            if isinstance(result, Exception):
