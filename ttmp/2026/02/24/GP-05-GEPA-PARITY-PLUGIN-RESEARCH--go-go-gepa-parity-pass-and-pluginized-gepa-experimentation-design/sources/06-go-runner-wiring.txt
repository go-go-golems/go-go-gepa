   279					Profile:       profile,
   280					EngineOptions: engineOptions,
   281				})
   282			})
   283		}
   284		if plugin.HasSelectComponents() {
   285			opt.SetComponentSelectorFunc(func(ctx context.Context, in gepaopt.ComponentSelectionInput) ([]string, error) {
   286				return plugin.SelectComponents(in, pluginEvaluateOptions{
   287					Profile:       profile,
   288					EngineOptions: engineOptions,
   289				})
   290			})
   291		}
   292		if plugin.HasComponentSideInfo() {
   293			opt.SetSideInfoFunc(func(ctx context.Context, in gepaopt.SideInfoInput) (string, error) {
   294				return plugin.ComponentSideInfo(in, pluginEvaluateOptions{
   295					Profile:       profile,
   296					EngineOptions: engineOptions,
   297				})
   298			})
   299		}
   300		if s.ShowEvents {
   301			opt.SetEventHook(func(event gepaopt.OptimizerEvent) {

   319		var recorder *runRecorder
   320		if s.Record {
   321			recordDB := strings.TrimSpace(s.RecordDB)

   324			}
   325			recorder, err = newRunRecorder(runRecorderConfig{
   326				DBPath:      recordDB,
   327				Mode:        "optimize",
   328				PluginID:    meta.ID,
   329				PluginName:  meta.Name,
   330				Profile:     profile,
   331				DatasetSize: len(examples),
   332				Objective:   s.Objective,
   333				MaxEvals:    s.MaxEvalCalls,
   334				BatchSize:   s.BatchSize,
   335				SeedPrompt:  seedTextForRecord,
   336			})
   337			if err != nil {
   338				return errors.Wrap(err, "failed to create run recorder")
   339			}
   340		}
   341		finalizeRun := func(runErr error) error {
   342			if recorder == nil {
   343				return runErr
   344			}
   345			closeErr := recorder.Close(runErr == nil, runErr)
   346			if runErr != nil {
   347				return runErr
   348			}
   349			return closeErr
   350		}

   352		res, err := opt.Optimize(ctx, seedCandidate, examples)
   353		if err != nil {

   356		if recorder != nil {

   358				return finalizeRun(err)
   359			}
   360		}

   372			}

   374				return finalizeRun(errors.Wrap(err, "failed to write out prompt"))
   375			}
   376			fmt.Fprintf(w, "Wrote best prompt to: %s\n", s.OutPrompt)
