     1	package gepa
     2	
     3	import "time"
     4	
     5	// Config controls the GEPA-style optimization loop.
     6	type Config struct {
     7		// Maximum number of evaluator calls. In prompt optimization, a “rollout”
     8		// usually corresponds to one (candidate, example) evaluation.
     9		MaxEvalCalls int
    10	
    11		// BatchSize is the number of examples sampled per iteration for mutation testing.
    12		BatchSize int
    13	
    14		// FrontierSize caps the number of candidates kept on the Pareto frontier
    15		// in the single-objective fallback path.
    16		FrontierSize int
    17	
    18		// RandomSeed controls stochastic selection and batching.
    19		// If 0, the optimizer will use time-based entropy.
    20		RandomSeed int64
    21	
    22		// ReflectionSystemPrompt is the system prompt used for the reflection LLM.
    23		ReflectionSystemPrompt string
    24	
    25		// ReflectionPromptTemplate must include "<curr_param>" and "<side_info>" placeholders.
    26		ReflectionPromptTemplate string
    27	
    28		// MergeProbability is the probability of attempting a merge (crossover) step
    29		// instead of a standard reflective mutation.
    30		//
    31		// 0 disables merging (default).
    32		MergeProbability float64
    33	
    34		// MergeSystemPrompt is the system prompt used for the merge proposer LLM.
    35		// If empty, the ReflectionSystemPrompt (or the reflector default) is used.
    36		MergeSystemPrompt string
    37	
    38		// MergePromptTemplate must include "<param_a>", "<param_b>",
    39		// "<side_info_a>", and "<side_info_b>" placeholders.
    40		MergePromptTemplate string
    41	
    42		// MergeScheduler controls when merge should be attempted.
    43		//
    44		// Supported values:
    45		//  - "probabilistic" (default): use MergeProbability each iteration.
    46		//  - "stagnation_due": maintain a merges_due counter that increases when no child is accepted,
    47		//    and spend due merge attempts when possible.
    48		//
    49		// Any other value falls back to "probabilistic".
    50		MergeScheduler string
    51	
    52		// MaxMergesDue caps the internal merges_due counter used by MergeScheduler=stagnation_due.
    53		MaxMergesDue int
    54	
    55		// OptimizableKeys optionally restricts which candidate keys are eligible for mutation/merge.
    56		//
    57		// If empty, the optimizer defaults to using the keys present in the seed candidate.
    58		//
    59		// This is the primary mechanism to support multi-parameter optimization (e.g., multiple
    60		// module prompts in a larger system).
    61		OptimizableKeys []string
    62	
    63		// ComponentSelector controls how many and which keys are updated per iteration.
    64		//
    65		// Supported values:
    66		//  - "round_robin" (default): update one key per iteration, cycling through OptimizableKeys.
    67		//  - "all": update all OptimizableKeys every iteration.
    68		//
    69		// Any other value falls back to "round_robin".
    70		ComponentSelector string
    71	
    72		// Objective is an optional natural-language description of what we are optimizing for.
    73		Objective string
    74	
    75		// MaxSideInfoChars caps the amount of formatted side-info passed to the reflector.
    76		// 0 means “no explicit cap”.
    77		MaxSideInfoChars int
    78	
    79		// Epsilon is the minimum improvement required to accept a child over its parent
    80		// in the single-objective setting. 0 is fine for most use.
    81		Epsilon float64
    82	
    83		// Now is injectable for tests. If nil, time.Now is used.
    84		Now func() time.Time
    85	}
    86	
    87	func (c Config) withDefaults() Config {
    88		out := c
    89		if out.MaxEvalCalls <= 0 {
    90			out.MaxEvalCalls = 200
    91		}
    92		if out.BatchSize <= 0 {
    93			out.BatchSize = 8
    94		}
    95		if out.FrontierSize <= 0 {
    96			out.FrontierSize = 10
    97		}
    98		if out.RandomSeed == 0 {
    99			out.RandomSeed = time.Now().UnixNano()
   100		}
   101		if out.ReflectionSystemPrompt == "" {
   102			out.ReflectionSystemPrompt = "You are an expert prompt engineer."
   103		}
   104		if out.ReflectionPromptTemplate == "" {
   105			out.ReflectionPromptTemplate = DefaultReflectionPromptTemplate
   106		}
   107		if out.MergeProbability < 0 {
   108			out.MergeProbability = 0
   109		}
   110		if out.MergeSystemPrompt == "" {
   111			out.MergeSystemPrompt = out.ReflectionSystemPrompt
   112		}
   113		if out.MergePromptTemplate == "" {
   114			out.MergePromptTemplate = DefaultMergePromptTemplate
   115		}
   116		if out.MergeScheduler == "" {
   117			out.MergeScheduler = "probabilistic"
   118		}
   119		if out.MaxMergesDue <= 0 {
   120			out.MaxMergesDue = 2
   121		}
   122		if out.ComponentSelector == "" {
   123			out.ComponentSelector = "round_robin"
   124		}
   125		if out.Now == nil {
   126			out.Now = time.Now
   127		}
   128		return out
   129	}
